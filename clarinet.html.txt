<script type="importmap">
    { "imports": { "smplr": "https://unpkg.com/smplr@0.9.0/dist/index.mjs" } }
</script>
<script type="module">
    import { Soundfont } from 'smplr';
    const audioCtx = new (window.AudioContext || window.webkitAudioContext)();
    let clarinet = null;

    async function loadClarinet() {
        if (clarinet) return clarinet;
        const sf = new Soundfont(audioCtx, { instrument: 'clarinet' });
        await sf.load;
        clarinet = sf;
        return clarinet;
    }

    async function playNote(midi) {
        if (audioCtx.state === 'suspended') await audioCtx.resume();
        const inst = await loadClarinet();
        if (!inst) return;
        inst.start({ note: midi, velocity: 90 });
        setTimeout(() => inst.stop({ note: midi }), 800);
    }

    const notes = [
        { midi: 64, label: 'E4' }, { midi: 66, label: 'F#4' },
        { midi: 67, label: 'G4' }, { midi: 69, label: 'A4' },
        { midi: 71, label: 'B4' }, { midi: 72, label: 'C5' },
        { midi: 74, label: 'D5' }, { midi: 76, label: 'E5' }
    ];

    const container = document.getElementById('keys');
    notes.forEach(n => {
        const btn = document.createElement('div');
        btn.className = 'key';
        btn.innerText = n.label;
        btn.onclick = () => playNote(n.midi);
        container.appendChild(btn);
    });

    document.body.addEventListener('click', () => { if (audioCtx.state === 'suspended') audioCtx.resume(); });
</script>
